const selectOpts = [
  { id: 1, value: 'Option One' },
  { id: 'test-string', value: 'Option Two' },
  { id: 3, value: 'Option Three' },
  { id: 4, value: 'Option Four' },
];

class ComponentExample extends React.Component {
  state = {
    selected: 'test-string',
  };

  handleSelect = (name, value) => {
    const newState = {};

    newState[name] = value;

    console.log(newState);
  }

  render() {
    return (
      <Select onSelect={this.handleSelect} selected={this.state.selected} name="mySelect" label="Choose One" options={selectOpts} />
    );
  }
}
