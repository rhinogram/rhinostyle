class TagInputComponentExample extends React.Component {
  state = {
    tags: ['Hello world!', 'Rhinostylee', 'O\'Doyle Rules!'],
    tagInputValue: '',
  };

  handleChange = (name, value) => {
    this.setState({ tagInputValue: value });
  };

  handleAddTag = () => {
    this.setState((prevState) => ({ tags: [...prevState.tags, prevState.tagInputValue], tagInputValue: '' }));
  }

  render() {
    return (
      <div className="form__group" style={{ display: 'flex' }}>
        <TagInput
          initialTagInputValue={this.state.tagInputValue}
          onChange={this.handleChange}
          tags={this.state.tags}
          tagCloseIconClassName="u-text-muted"
          name="myTags"
        />
        <div>
          <Button className="u-m-l" size="large" type="secondary" onClick={this.handleAddTag}>Add New Tag</Button>
        </div>
      </div>
    );
  }
}
